package middleware

import (
	"log"
	"net/http"
)

type Middleware func(http.Handler) http.Handler

func Handle() Middleware {
	return func(h http.Handler) http.Handler {
		return Logger()(h) //TODO: chain your middleware http.handlers
	}
}
